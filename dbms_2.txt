1: DECLARE
    dup_row EXCEPTION;
    PRAGMA EXCEPTION_INIT(dup_row, -12345);
    s_row student%rowtype;
    CURSOR s_cur IS SELECT * FROM student;
    r number;
    n VARCHAR2(50);
    m number;
BEGIN
r:=:Roll;
n:=:Name;
m:=:Mark;
OPEN s_cur;
LOOP 
FETCH s_cur INTO s_row;
    CASE 
        WHEN r = s_row.roll THEN 
            RAISE dup_row;
        WHEN n = s_row.name THEN 
            RAISE dup_row;
        ELSE 
            dbms_output.put_line('valid data');                
    END CASE;                
END LOOP;
EXCEPTION
    WHEN dup_row THEN
        dbms_output.put_line('Error: Same student already Exists!! Code: ' || SQLCODE);               
    WHEN OTHERS THEN
        dbms_output.put_line('An unexpected error occurred: ' || SQLERRM);
END;


2: DECLARE
    min_sal_excep EXCEPTION;
    max_sal_excep EXCEPTION;
    PRAGMA EXCEPTION_INIT(max_sal_excep, -12345);
    PRAGMA EXCEPTION_INIT(min_sal_excep, -12344); 
    max_sal number;
    min_sal number;
    deptno number;               
    CURSOR emp_cur IS SELECT deptno,MIN(sal), MAX(sal) FROM emp GROUP BY deptno;
    
BEGIN
OPEN emp_cur;
LOOP 
FETCH emp_cur INTO deptno,min_sal,max_sal;
    CASE 
        WHEN min_sal < 1000 THEN 
            RAISE min_sal_excep;
        WHEN max_sal > 4000 THEN 
            RAISE max_sal_excep;
        ELSE 
            dbms_output.put_line('valid data');                
    END CASE;                
END LOOP;
CLOSE emp_cur;
EXCEPTION
    WHEN max_sal_excep THEN
        dbms_output.put_line('Error: Maximum Salary reached!! Code: ' || SQLCODE); 
        dbms_output.put_line('DEPT:'||(deptno));
    WHEN min_sal_excep THEN
        dbms_output.put_line('Error: Minimum Salary reached!! Code: ' || SQLCODE);
        dbms_output.put_line('DEPT:'||(deptno));              
    WHEN OTHERS THEN
        dbms_output.put_line('An unexpected error occurred: ' || SQLERRM);
END;


3: DECLARE
    low_employee EXCEPTION;
    PRAGMA EXCEPTION_INIT(low_employee, -12345);
    dno number;
    name VARCHAR2(50);               
    CURSOR emp_cur IS SELECT ename,deptno FROM emp;
    
BEGIN
OPEN emp_cur;
LOOP 
FETCH emp_cur INTO name,dno;
    CASE 
        WHEN dno < 4 THEN 
            RAISE low_employee;
        ELSE 
            dbms_output.put_line('valid data');                
    END CASE;                
END LOOP;
CLOSE emp_cur;
EXCEPTION
    WHEN low_employee THEN
        dbms_output.put_line('Error: Invalid Dept Number!! Code: ' || SQLCODE); 
        dbms_output.put_line('DEPT:'||(dno)||'Name:'||(name));            
    WHEN OTHERS THEN
        dbms_output.put_line('An unexpected error occurred: ' || SQLERRM);
END;

8: DECLARE
    no_employee EXCEPTION;
    PRAGMA EXCEPTION_INIT(no_employee, -12345);
    name VARCHAR2(50);
    dname VARCHAR2(50);
    dno number;              
    CURSOR emp_cur IS SELECT ename,d.deptno,dname FROM emp e RIGHT OUTER JOIN dept d ON (e.deptno=d.deptno);
BEGIN
OPEN emp_cur;
LOOP 
FETCH emp_cur INTO name,dno,dname;
    EXIT WHEN emp_cur%NOTFOUND;
    CASE 
        WHEN name IS NULL THEN 
            RAISE no_employee;
        ELSE 
            dbms_output.put_line('Name:'||name);
            dbms_output.put_line('-----------');   
    END CASE;  
                  
END LOOP;
CLOSE emp_cur;
EXCEPTION

    WHEN no_employee THEN
        dbms_output.put_line('Error:No employee exists!! Code: ' || SQLCODE);  
        dbms_output.put_line('DEPT:'||(dno));        
    WHEN OTHERS THEN
        dbms_output.put_line('An unexpected error occurred: ' || SQLERRM);
END;

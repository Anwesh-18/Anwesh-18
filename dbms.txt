Q4: declare
large number;

e_name VARCHAR2(50);
jb VARCHAR2(30);
mgr number;
hiredate DATE;
salary number;

Q6. declare
e1_name VARCHAR2(50);
e2_name VARCHAR2(50);
begin
SELECT ename INTO e1_name FROM emp WHERE sal=(SELECT MAX(sal) FROM emp);
SELECT ename INTO e2_name FROM emp WHERE sal=(SELECT MIN(sal) FROM emp);
dbms_output.put_line('Highest_salary: '||(e1_name));
dbms_output.put_line('Lowest_salary: '||(e2_name));
end;
